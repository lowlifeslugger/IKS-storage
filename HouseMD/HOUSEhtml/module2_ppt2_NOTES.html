Comments in Python


# This is a single-line comment (starts with #)
# Everything after # on the same line is a comment
# Comments are non-executable — compiler and PVM skip them


"""
This is a multiline string used as a comment.
(Triple double quotes)
"""

'''
This is also a multiline string.
(Triple single quotes)
'''

> Important: Triple-quoted strings are NOT true comments — they are regular strings that can span multiple lines. Using them as comments wastes interpreter time by consuming memory. Use # for true comments.


---


Docstrings


When a triple-quoted string is written as the first statement in a module, function, or method, it becomes a docstring (documentation string).


- Docstrings are used to create API documentation files (text or HTML).

- API doc files describe all features of a software, language, or product.


def add(a, b):
    """Returns the sum of a and b."""
    return a + b

Variables and Identifiers


# In C/Java — explicit type declaration required:
# int a;
# a = 10;

# In Python — no declaration needed:
a = 10        # Python automatically determines type
b = 15

Identifier — A name given to a variable, function, class, etc.


Rules:

- Can include letters, numbers, and underscore _.

- Must start with a non-numeric character.

- Special symbols (?, #, $, %, @) are NOT allowed.

- Case-sensitive: Num and num are different identifiers.


Constants — Indicated by ALL_CAPS (convention only, not enforced by Python):


MAX_SIZE = 100
PI = 3.14159

1. NoneType


Represents an object with no value. Used as default argument value in functions. Evaluates to False in Boolean context.


def greet(name=None):
    if name is None:
        print("Hello, Stranger!")
    else:
        print(f"Hello, {name}!")

greet()          # Output: Hello, Stranger!
greet("Alice")   # Output: Hello, Alice!

Type Conversion (Explicit)


# Basic conversions
int(3.7)          # → 3     (truncates, doesn't round)
float(5)          # → 5.0
complex(3)        # → (3+0j)
complex(3, 5)     # → (3+5j)

# Number system conversions
bin(10)           # → '0b1010'    (decimal to binary string)
oct(10)           # → '0o12'     (decimal to octal string)
hex(10)           # → '0xa'      (decimal to hex string)

# String with base to decimal
int('1010', 2)    # binary string  → 10
int('A', 16)      # hex string     → 10
int('12', 8)      # octal string   → 10
int('5', 8)       # octal 5        → 5
int('0010', 2)    # binary 0010    → 2
int('E', 16)      # hex E          → 14

3. Sequences


str (String)


s1 = 'welcome to python'
s2 = "welcome to python"
s3 = """multiline
string"""

# Embedding quotes
s4 = 'welcome to "Core Python" learning'
s5 = "welcome to 'Core Python' learning"

# Indexing — starts from 0
s1[0]       # 'w'   (first character)
s1[-1]      # 'n'   (last character)

# Slicing — [start : stop : step]
s1[0:5]     # 'welco'
s1[::-1]    # reverse entire string
s1[::2]     # every other character

# Repetition
s1 * 2      # repeat string twice

# Concatenation
"hello" + " world"   # 'hello world'

# String operations — string examples
s = 'hello world'
print(s[0])        # 'h'
print(s[:5])       # 'hello'
print(s[6:])       # 'world'
print(s[-1])       # 'd'
print(s[::-1])     # 'dlrow olleh'

---


bytes


# Group of byte numbers (0–255); IMMUTABLE — cannot modify elements
b = bytes([10, 20, 30, 40])

print(b[0])     # 10
print(b[1])     # 20
# b[0] = 99    # ERROR — bytes are immutable

---


bytearray


# Same as bytes but MUTABLE — can modify elements
ba = bytearray([10, 20, 0, 40, 15])

ba[0] = 99     # allowed
ba[1] = 88     # allowed
print(ba)      # bytearray(b'X\x00(..)')

---


list


# MUTABLE; can store MIXED types; dynamic size; uses [ ]
lst = [10, 20, "abc", "xyz"]

print(lst)        # [10, 20, 'abc', 'xyz']
print(lst[0])     # 10        (first)
print(lst[0:2])   # [10, 20]  (first two)
print(lst[1:3])   # [20, 'abc'] (second and third)
print(lst[-1])    # 'xyz'     (last)
print(lst * 2)    # list repeated twice

---


tuple


# IMMUTABLE — read-only list; uses ( )
tp = (10, 20, "abc", "xyz")

print(tp)         # (10, 20, 'abc', 'xyz')
print(tp[0])      # 10
print(tp[0:2])    # (10, 20)
print(tp[-1])     # 'xyz'
print(tp * 2)     # tuple repeated twice

# tp[0] = 99      # TypeError — cannot modify tuple elements

---


range


# Represents a sequence of numbers
r1 = range(10)           # 0, 1, 2, ..., 9
r2 = range(30, 40, 2)    # 30, 32, 34, 36, 38

# Display using for loop
for i in range(10):
    print(i)

# Create list from range
lst = list(range(5))     # [0, 1, 2, 3, 4]

4. Sets


set


# UNORDERED; NO DUPLICATES; MUTABLE; uses { }
s = {10, 20, 30, 50}

s.update([60, 70])    # add elements
s.remove(20)          # remove element

# s[0]   # ERROR — sets are unordered; no indexing

---


frozenset


# IMMUTABLE set — cannot add or remove elements
s = {10, 20, 30, 50}
fs = frozenset(s)
print(fs)             # frozenset({10, 20, 30, 50})

# Create from string
fs2 = frozenset("abcdefg")
print(fs2)            # frozenset({'a', 'b', 'c', 'd', 'e', 'f', 'g'})

# fs.update(...)      # ERROR — frozenset is immutable
# fs.remove(...)      # ERROR

5. Mappings — dict


# KEY-VALUE pairs; colon (:) separates key and value; uses { }
d = {10: 'Rohan', 11: 'Krishna', 12: 'Ram'}

# Access value by key
print(d[10])          # 'Rohan'

# Get all keys and values
print(d.keys())       # dict_keys([10, 11, 12])
print(d.values())     # dict_values(['Rohan', 'Krishna', 'Ram'])

# Update a value
d[10] = 'Raj'

# Delete a key-value pair
del d[11]

# Create empty dict and add entries
d = {}
d[1] = 'Alice'
d[2] = 'Bob'
print(d)              # {1: 'Alice', 2: 'Bob'}

Literals in Python


A literal is a constant value stored directly into a variable.


# Numeric literals
a = 10          # integer literal
b = 3.14        # float literal

# Boolean literals
c = True
d = False

# String literals
s1 = 'hello'           # single quotes
s2 = "world"           # double quotes
s3 = '''multiline
string'''              # triple quotes — spans multiple lines
s4 = """also
multiline"""

# Multiline with backslash continuation
s5 = "this is a long " \
     "string joined"

Determining Datatype


a = 10
b = 3.14
c = 3 + 5j
d = True
e = "hello"
f = [1, 2, 3]

print(type(a))    # <class 'int'>
print(type(b))    # <class 'float'>
print(type(c))    # <class 'complex'>
print(type(d))    # <class 'bool'>
print(type(e))    # <class 'str'>
print(type(f))    # <class 'list'>

# Python has no 'char' datatype — individual characters are str
ch = 'B'
print(type(ch))   # <class 'str'>
print(ch == 'B')  # True
print(ch == 'b')  # False
