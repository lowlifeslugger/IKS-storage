What is a String?


A string represents a group of characters. Most real-world data — names, addresses, vehicle numbers, credit card numbers — are strings.


In Python, the str datatype represents strings. There is no separate char type — individual characters are also str.

Creating Strings


# Single quotes and double quotes are interchangeable
s1 = 'welcome to Core Python learning'
s2 = "welcome to Core Python learning"

# Triple quotes — spans multiple lines
s3 = """Welcome to Core Python,
a book on Python language that discusses
all important concepts in a lucid manner."""

s4 = '''Also works with
triple single quotes.'''

# Embedding quotes — use opposite quote types
s5 = 'welcome to "Core Python" learning'   # single outer, double inner
s6 = "welcome to 'Core Python' learning"   # double outer, single inner

print(s5)    # welcome to "Core Python" learning
print(s6)    # welcome to 'Core Python' learning

Raw Strings


Add r before the string to treat escape characters as literal characters (no effect applied).


s = r"Welcome to\tCore Python\nlearning"
print(s)
# Output: Welcome to\tCore Python\nlearning
# (no tab or newline — shown as-is)

Indexing in Strings


s = 'Core Python'
     0123456789 10     ← positive indices
    -11      ...  -1   ← negative indices


s = 'Core Python'

print(s[0])     # 'C'   — first character
print(s[1])     # 'o'
print(s[5])     # 'P'
print(s[-1])    # 'n'   — last character
print(s[-2])    # 'o'
print(s[-6])    # 'P'   (6th from end)

# Access each character using while loop (forward)
n = len(s)
i = 0
while i < n:
    print(s[i], end=' ')
    i += 1

print()   # new line

# Access in reverse using while loop
i = -1
while i >= -n:
    print(s[i], end=' ')
    i -= 1

print()

# Using for loop
for i in s:
    print(i, end=' ')

print()

# Reverse using for loop and slicing
for i in s[::-1]:
    print(i, end=' ')

Slicing Strings


Format: stringname[start : stop : stepsize]


- If start is omitted → defaults to 0.

- If stop is omitted → defaults to last character.

- If stepsize is omitted → defaults to 1.


s = 'Core Python'
#    0123456789 10

s[0:9:1]      # 'Core Pyth'   — from 0 to 8, step 1
s[0:9:2]      # 'Cr yh'       — from 0 to 8, step 2
s[::]         # 'Core Python' — entire string
s[2:4:1]      # 're'          — from index 2 to 3
s[::2]        # 'Cr yhn'      — every other character
s[2::]        # 're Python'   — from index 2 to end
s[:4:1]       # 'Core'        — from 0 to 3

# Negative indexing in slicing
s[-4:-1]      # 'tho'         — from -4 to -2
s[-6::]       # 'Python'      — from -6 to end
s[-1:-4:-1]   # 'noh'         — right to left
s[-1::-1]     # 'nohtyP eroC' — full reverse

Checking Membership


s = 'This is Core Python'

# in — returns True if found
if 'Core' in s:
    print("Core is found in main string")

# not in — returns True if NOT found
if 'Java' not in s:
    print("Java is not found")

# Program — check user-entered substring
str_main = input('Enter main string: ')
sub = input('Enter sub string: ')

if sub in str_main:
    print(sub + ' is found in main string')
else:
    print(sub + ' is not found in the main string')

> Note: in and not in are case-sensitive. 'core' ≠ 'Core'.

Comparing Strings


Use relational operators: ==, !=, <, >, <=, >=.


Python compares strings in dictionary (lexicographic) order — 'A' < 'B' < 'C' ... < 'a' < 'b' ...


s1 = 'Box'
s2 = 'Boy'

if s1 == s2:
    print("Both are same")
elif s1 < s2:
    print("s1 comes before s2")    # Output: s1 comes before s2
else:
    print("s1 comes after s2")

# 'Box' < 'Boy' because at index 2: 'x' < 'y' in ASCII

Removing Spaces


name = '  Mukesh Deshmukh  '   # spaces before and after

print(name.rstrip())    # '  Mukesh Deshmukh'   — removes RIGHT spaces
print(name.lstrip())    # 'Mukesh Deshmukh  '   — removes LEFT spaces
print(name.strip())     # 'Mukesh Deshmukh'     — removes BOTH sides

# These do NOT remove spaces in the MIDDLE of the string

Finding Substrings


Method Description
find(sub, start, end) Returns index of first occurrence; -1 if not found
rfind(sub, start, end) Same, but searches from RIGHT to LEFT
index(sub, start, end) Returns index; raises ValueError if not found
rindex(sub, start, end) Same, but searches from RIGHT to LEFT 

s = 'This is a book'

# find() — returns index or -1
n = s.find('is', 0, len(s))
print(n)    # 2   (first 'is' at index 2)

# rfind() — search from right
n = s.rfind('is')
print(n)    # 5   (second 'is' at index 5)

# index() — raises ValueError if not found
try:
    n = s.index('is', 0, len(s))
    print(n)    # 2
except ValueError:
    print("Not found")

# Program — find first occurrence
str_main = input('Enter main string: ')
sub = input('Enter sub string: ')
n = str_main.find(sub, 0, len(str_main))

if n == -1:
    print("Sub string not found")
else:
    print("Sub string found at position:", n + 1)   # +1 for 1-based position

Finding All Occurrences


# Version 1 — using flag and manual logic
str_main = input('Enter main string: ')
sub = input('Enter sub string: ')
i = 0
n = len(str_main)
flag = False

while i < n:
    pos = str_main.find(sub, i, n)
    if pos != -1:
        print("Found at position:", pos + 1)
        flag = True
        i = pos + 1       # continue search from next position
    else:
        i += 1

if not flag:
    print("Sub string not found")

# Version 2 — cleaner (using pos = -1 trick)
str_main = input('Enter main string: ')
sub = input('Enter sub string: ')
pos = -1
n = len(str_main)
flag = False

while True:
    pos = str_main.find(sub, pos + 1, n)
    if pos == -1:
        break
    print("Found at position:", pos + 1)
    flag = True

if not flag:
    print("Not found")

Counting Substrings


# count(substring) — returns number of occurrences
str_main = 'New Delhi is in Delhi'

print(str_main.count('Delhi'))              # 2   — count in entire string
print(str_main.count('e', 0, 3))           # 1   — count in range 0 to 2
print(str_main.count('e', 0, len(str_main)))  # 3   — count in full string

Strings are Immutable


Strings cannot be modified — any attempt to change a character raises a TypeError.


s = 'abcd'
print(s[0])       # 'a'

# s[0] = 'x'     # TypeError: 'str' object does not support item assignment

# What LOOKS like modification is actually creating a new object
s1 = 'one'
s2 = 'two'

s2 = s1           # s2 now POINTS to 'one'; 'two' is garbage collected
print(s2)         # 'one'
print(s1)         # 'one'
print(id(s1) == id(s2))    # True — both refer to same object

Why immutable?

- Performance — Fixed size in memory at creation; faster allocation and access.

- Security — Modifications create new objects with new identity, so tampering is detectable.

Replacing a String

Splitting and Joining Strings


split() — Break string into list


# Split at commas
s = 'one, two, three, four'
lst = s.split(',')
print(lst)    # ['one', ' two', ' three', ' four']

# Split at spaces (default — splits at any whitespace)
s = 'hello world python'
words = s.split()
print(words)  # ['hello', 'world', 'python']

# Program — accept numbers separated by space
str_input = input('Enter numbers separated by space: ')
lst = str_input.split()
for i in lst:
    print(i)
# Input: 10 20 30 40
# Output: 10  20  30  40

join() — Combine list/tuple into single string


# Format: separator.join(sequence)

# Tuple of strings joined with hyphen
words = ('one', 'two', 'three')
result = '-'.join(words)
print(result)    # one-two-three

# List joined with colon
cities = ['apple', 'guava', 'grapes', 'mango']
result = ':'.join(cities)
print(result)    # apple:guava:grapes:mango

Changing Case


s = 'Python is the future'

print(s.upper())        # PYTHON IS THE FUTURE
print(s.lower())        # python is the future
print(s.swapcase())     # pYTHON IS THE FUTURE  (swap each letter's case)
print(s.title())        # Python Is The Future   (capitalize first letter of each word)

String Testing Methods


Return True or False based on content.


Method Returns True if...
isalnum() All characters are alphanumeric (A–Z, a–z, 0–9)
isalpha() All characters are alphabetic
isdigit() All characters are digits
islower() All characters are lowercase
isupper() All characters are uppercase
isspace() All characters are whitespace
istitle() String is in title case 

s1 = 'Python3'
s2 = '12345'
s3 = 'hello'
s4 = 'HELLO'
s5 = 'Hello World'

print(s1.isalnum())    # True  — only letters and digits
print(s1.isalpha())    # False — has digit '3'
print(s2.isdigit())    # True
print(s3.islower())    # True
print(s4.isupper())    # True
print(s5.istitle())    # True  — each word starts with capital
print('  '.isspace())  # True
